  <Story name='基础用法' height='420px'>
    {{
      components: { Coil },
      template: `<div>
        <Coil :width="400" :height="400" ref="coil" src="http://img.ikstatic.cn/MTU5MDU0NDQyMTA3MiM5NzAjanBn.jpg"/>
        <div>
          <button v-for="(it,i) in btns" 
            :key="i" 
            type="button" 
            @click="addShape(it.name)" 
            :style="{background:state === it.name ? '#ccc' :'#fff'}"
            >
              {{it.label}}
          </button>
        </div>
        <div>快捷键</div>
        <p>Ctrl+z、Ctrl+y 历史记录控制</p>
        <p>Ctrl+c、Ctrl+v 、Ctrl+x、Ctrl+backspace 复制、删除</p>
        <p>拖动角+Shift 矩形、圆形正规化拖拽</p>
        <p>拖动角+Ctrl 保持比例拖拽</p>
        <p>拖动角+alt 快速复制 并拖动</p>
      </div>`,
      data () {
        return {
          state:null,
          btns:[
            {
              label:'多边形',
              name:'polygon'
            },
            {
              label:'矩形',
              name:'rect'
            },
            {
              label:'圆形',
              name:'ellipse'
            },
            {
              label:'打点',
              name:'dot'
            }
          ]
        }
      },
      methods:{
        addShape(type){
          const coil = this.$refs.coil
          let t = null
          if(this.state !== type){
            t = type
          }
          this.state = t
          coil.$setState(t)
        }
      },
      mounted (){
      }
    }}
  </Story>
