<Preview>
  <Story name='基础用法' height='420px'>
    {{
      components: { Coil },
      template: `<div>
        <Coil @selectTool="selectTool" @changeBlocks="changeBlocks" :width="400" :height="400" ref="coil" src="http://img.ikstatic.cn/MTU5MDU0NDQyMTA3MiM5NzAjanBn.jpg" :ctrln="ctrln"/>
        <pre style="position: absolute;right: 0;top: 0;">{{info}}</pre>
        <div>
          <button v-for="(it,i) in btns" 
            :key="i" 
            type="button" 
            @click="addShape(it.name)" 
            :style="{background:state === it.name ? '#ccc' :'#fff'}"
            >
              {{it.label}}
          </button>
        </div>
        <div>快捷键</div>
        <p>Ctrl+z、Ctrl+y 历史记录控制</p>
        <p>Ctrl +、Ctrl - 控制缩放</p>
        <p>Ctrl+c、Ctrl+v 、Ctrl+x、Ctrl+d 复制、删除</p>
        <p>Ctrl+1、Ctrl+2 、Ctrl+3、Ctrl+4 选中对应工具</p>
        <p>拖动角+Shift 矩形、圆形正规化拖拽</p>
        <p>拖动角+Ctrl 保持比例拖拽</p>
        <p>拖动角+alt 快速复制 并拖动</p>
      </div>`,
      data () {
        return {
          ctrln:['polygon','rect','ellipse','dot'],
          state:null,
          info:[],
          btns:[
            {
              label:'多边形',
              name:'polygon'
            },
            {
              label:'矩形',
              name:'rect'
            },
            {
              label:'圆形',
              name:'ellipse'
            },
            {
              label:'打点',
              name:'dot'
            }
          ]
        }
      },
      methods:{
        changeBlocks(v){
          console.log(v)
          const info = v.map(({type,paths,rotate})=>{
            console.log(type)
            let o = {}
            if(type==='ellipse'){
              o = {type,data:JSON.stringify(this.path2ellipse(paths,rotate))}
            }else if(type==='rect'){
              o = {type,data:JSON.stringify(this.path2rect(paths,rotate))}
            }else {
              o = {type,data:JSON.stringify(paths)}
            }
            return o
          })
          this.info = JSON.stringify(info,null,2)
        },
        path2ellipse(paths,rotate){
          return {
            cx:parseInt((paths[0][0]+paths[3][0])/2), 
            cy:parseInt((paths[0][1]+paths[1][1])/2),
            rx:parseInt((paths[3][0]-paths[0][0])/2),
            ry:parseInt((paths[1][1]-paths[0][1])/2),
            rotate:parseInt(rotate)
          }
        },
        path2rect(paths,rotate){
          return {
            left:parseInt(paths[0][0]),
            top:parseInt(paths[0][1]),
            width:parseInt(paths[3][0]-paths[0][0]),
            height:parseInt(paths[1][1]-paths[0][1]),
            rotate:parseInt(rotate)
          }
        },
        selectTool(t){
          this.state = t
        },
        addShape(type){
          const coil = this.$refs.coil
          let t = null
          if(this.state !== type){
            t = type
          }
          this.state = t
          coil.$setState(t)
        }
      },
      mounted (){
      }
    }}
  </Story>
